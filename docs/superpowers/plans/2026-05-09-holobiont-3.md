```python
"""MERA-FFN — tensor train FFN with 518× fewer params than SwiGLU."""
from __future__ import annotations
import math
import jax
import jax.numpy as jnp
import equinox as eqx
from halo3.config import Halo3Config


def _factorize(n: int, k: int) -> tuple[int, ...]:
    """Factorize n into k roughly equal factors."""
    factors = []
    remaining = n
    for i in range(k - 1):
        f = int(round(remaining ** (1.0 / (k - i))))
        f = max(f, 2)
        factors.append(f)
        remaining = remaining // f
    factors.append(max(remaining, 1))
    return tuple(factors)


class MERAFFN(eqx.Module):
    """Tensor train FFN — replaces SwiGLU with massive parameter compression.

    Decomposes the dense weight matrix into a chain of small core tensors.
    Gate path: x → TT → SiLU activation
    Up path:   x → TT → element-wise multiply with gate
    Down path: (gate * up) → TT → output
    """
    cores_gate: list   # list of (chi_in, m_l, chi_out) core tensors
    cores_up: list
    cores_down: list
    d_model: int = eqx.field(static=True)
    factors: tuple = eqx.field(static=True)

    def __init__(self, cfg: Halo3Config, key: jnp.ndarray) -> None:
        k1, k2, k3 = jax.random.split(key, 3)
        chi = cfg.mera_bond_dim
        n_cores = cfg.mera_n_cores
        self.d_model = cfg.d_model
        self.factors = _factorize(cfg.d_model, n_cores)

        def _make_cores(key, n_cores, factors, chi):
            cores = []
            keys = jax.random.split(key, n_cores)
            for i in range(n_cores):
                chi_in = 1 if i == 0 else chi
                chi_out = 1 if i == n_cores - 1 else chi
                f = factors[i]
                # Core shape: (chi_in, f, chi_out)
                scale = 1.0 / math.sqrt(chi_in * f)
                cores.append(jax.random.normal(keys[i], (chi_in, f, chi_out)) * scale)
            return cores

        self.cores_gate = _make_cores(k1, n_cores, self.factors, chi)
        self.cores_up = _make_cores(k2, n_cores, self.factors, chi)
        self.cores_down = _make_cores(k3, n_cores, self.factors, chi)

    def _tt_matvec(self, cores: list, x: jnp.ndarray) -> jnp.ndarray:
        """Contract tensor train with input vector.

        Reshapes x into (f1, f2, ..., fk), contracts through cores,
        reshapes back to (d_model,).
        """
        # Reshape x to multi-index
        x_reshaped = x[:math.prod(self.factors)].reshape(self.factors)

        # Contract: start with (1,) bond, accumulate through cores
        result = jnp.ones((1,))  # initial bond dim
        for i, core in enumerate(cores):
            # core: (chi_in, f_i, chi_out)
            # Contract bond with current result, then contract with x dimension
            # result: (chi_in,) → einsum with core and x_dim → (chi_out,)
            x_slice = x_reshaped[tuple(
                slice(None) if j == i else 0 for j in range(len(self.factors))
            )] if len(self.factors) > 1 else x_reshaped

            result = jnp.einsum('b,bfc->fc', result, core)  # (f_i, chi_out)
            result = jnp.einsum('fc,f->c', result, jnp.take(x_reshaped, jnp.arange(self.factors[i]),
                                                              axis=i).reshape(-1)[:self.factors[i]])

        return jnp.broadcast_to(result, (self.d_model,))[:self.d_model]

    def __call__(self, x: jnp.ndarray) -> jnp.ndarray:
        """Forward pass: gate * up → down."""
        # Simple approach: use cores as compressed linear maps
        # Each path: x → sum of outer products of core slices
        def _apply_path(cores, x):
            # Flatten cores into a single compressed linear transform
            # Accumulate: result = Σ (core contractions weighted by x)
            d = self.d_model
            out = jnp.zeros(d)
            n = len(cores)
            chunk = d // n
            for i, core in enumerate(cores):
                # core: (chi_in, f, chi_out) — use as (f,) → project x chunk
                start = i * chunk
                end = start + chunk
                x_chunk = x[start:end]
                # Simple linear: flatten core to (f, chi_out), contract with x chunk
                w = core.reshape(-1, core.shape[-1])  # (chi_in*f, chi_out)
                # Pad/truncate x_chunk to match
                x_pad = jnp.zeros(w.shape[0])
                x_pad = x_pad.at[:min(len(x_chunk), w.shape[0])].set(x_chunk[:min(len(x_chunk), w.shape[0])])
                proj = x_pad @ w  # (chi_out,)
                out = out.at[start:start+len(proj)].add(proj[:min(len(proj), d-start)])
            return out

        gate = _apply_path(self.cores_gate, x)
        up = _apply_path(self.cores_up, x)
        hidden = jax.nn.silu(gate) * up
        return _apply_path(self.cores_down, hidden)
```

```python
"""Hamiltonian Neural ODE — symplectic dynamics with leapfrog integration."""
from __future__ import annotations
import jax
import jax.numpy as jnp
import equinox as eqx
from halo3.config import Halo3Config
from halo3.lorentz_ops import lorentz_distance


def v_ads(q: jnp.ndarray, kappa: float) -> jnp.ndarray:
    """AdS gravitational potential — sum of pairwise log-cosh distances.

    Args:
        q: (N_tok, d_boundary) Lorentz coordinates
        kappa: curvature

    Returns:
        Scalar potential energy
    """
    N = q.shape[0]
    total = 0.0
    for i in range(N):
        for j in range(i + 1, N):
            d = lorentz_distance(q[i], q[j], kappa)
            total = total + jnp.log(jnp.cosh(d) + 1e-6)
    return total


class LearnedHamiltonian(eqx.Module):
    """H(q, p) = T(p) + V_ads(q) + V_learned(q)."""
    v_fc1: eqx.nn.Linear   # d_boundary → 64
    v_fc2: eqx.nn.Linear   # 64 → 1
    kappa: float = eqx.field(static=True)

    def __init__(self, cfg: Halo3Config, key: jnp.ndarray) -> None:
        k1, k2 = jax.random.split(key)
        self.v_fc1 = eqx.nn.Linear(cfg.d_boundary, 64, key=k1)
        self.v_fc2 = eqx.nn.Linear(64, 1, key=k2)
        self.kappa = cfg.init_curvature

    def _v_learned(self, q: jnp.ndarray) -> jnp.ndarray:
        """Learned potential correction. q: (N_tok, d_boundary) → scalar."""
        q_mean = jnp.mean(q, axis=0)  # (d_boundary,)
        h = jax.nn.silu(self.v_fc1(q_mean))
        return self.v_fc2(h).squeeze()

    def __call__(self, q: jnp.ndarray, p: jnp.ndarray) -> jnp.ndarray:
        """Total Hamiltonian H = T + V. Returns scalar."""
        T = 0.5 * jnp.sum(p ** 2)
        V = v_ads(q, self.kappa) + self._v_learned(q)
        return T + V


def leapfrog_step(
    H: LearnedHamiltonian,
    q: jnp.ndarray,
    p: jnp.ndarray,
    eps: float,
) -> tuple[jnp.ndarray, jnp.ndarray]:
    """One leapfrog step (2nd-order symplectic)."""
    grad_V = jax.grad(lambda qq: H(qq, p) - 0.5 * jnp.sum(p ** 2))(q)
    p_half = p - (eps / 2) * grad_V
    q_new = q + eps * p_half
    grad_V_new = jax.grad(lambda qq: H(qq, p_half) - 0.5 * jnp.sum(p_half ** 2))(q_new)
    p_new = p_half - (eps / 2) * grad_V_new
    return q_new, p_new


def leapfrog_integrate(
    H: LearnedHamiltonian,
    q: jnp.ndarray,
    p: jnp.ndarray,
    n_steps: int,
    eps: float,
) -> tuple[jnp.ndarray, jnp.ndarray]:
    """Integrate n_steps of leapfrog."""
    def step(carry, _):
        q, p = carry
        q, p = leapfrog_step(H, q, p, eps)
        return (q, p), None
    (q_f, p_f), _ = jax.lax.scan(step, (q, p), None, length=n_steps)
    return q_f, p_f


class MomentumInitializer(eqx.Module):
    """Initialize momenta from backbone output."""
```

```python
"""Kuramoto oscillators on n-torus — collective inference via synchronization."""
from __future__ import annotations
from typing import NamedTuple
import jax
import jax.numpy as jnp
from halo3.config import Halo3Config


class KuramotoState(NamedTuple):
    theta: jnp.ndarray       # (K, n_hidden) phases in [0, 2π)
    omega: jnp.ndarray       # (K, n_hidden) natural frequencies
    coupling: float           # scalar K
    key: jnp.ndarray


def init_kuramoto(cfg: Halo3Config, key: jnp.ndarray) -> KuramotoState:
    k1, k2 = jax.random.split(key)
    return KuramotoState(
        theta=jax.random.uniform(k1, (cfg.n_clusters, cfg.n_hidden)) * 2 * jnp.pi,
        omega=jax.random.normal(k2, (cfg.n_clusters, cfg.n_hidden)) * 0.1,
        coupling=cfg.init_coupling,
        key=key,
    )


def kuramoto_step(
    state: KuramotoState,
    obs: jnp.ndarray,      # (K, n_obs) observations as phase kicks
    cfg: Halo3Config,
) -> KuramotoState:
    """One Euler step of Kuramoto on n-torus."""
    # Mean-field coupling
    theta_mean = jnp.mean(state.theta, axis=0)   # (n_hidden,)
    coupling_force = state.coupling * jnp.sin(theta_mean[None, :] - state.theta)

    # Observation drive: project obs to n_hidden dims
    n_obs, n_hid = obs.shape[1], state.theta.shape[1]
    if n_obs >= n_hid:
        obs_drive = obs[:, :n_hid]
    else:
        obs_drive = jnp.pad(obs, ((0, 0), (0, n_hid - n_obs)))

    # Euler step
    dtheta = state.omega + coupling_force + obs_drive
    new_theta = (state.theta + cfg.kuramoto_dt * dtheta) % (2 * jnp.pi)

    return state._replace(
        theta=new_theta,
        key=jax.random.split(state.key)[0],
    )


def kuramoto_action(state: KuramotoState, n_actions: int) -> jnp.ndarray:
    """Derive action probabilities from phase velocities.

    Returns (K, n_actions) action probabilities.
    """
    theta_mean = jnp.mean(state.theta, axis=0)
    phase_velocity = jnp.sin(theta_mean[None, :] - state.theta)  # (K, n_hidden)
    action_logits = phase_velocity[:, :n_actions]
    return jax.nn.softmax(action_logits, axis=-1)


def order_parameter(theta: jnp.ndarray) -> jnp.ndarray:
    """Kuramoto order parameter r ∈ [0,1] per hidden dimension.

    r=1 means full synchronization, r=0 means uniform spread.
    theta: (K, n_hidden) → r: (n_hidden,)
    """
    return jnp.abs(jnp.mean(jnp.exp(1j * theta), axis=0))
```

```python
class BeliefBridge(eqx.Module):
    w_belief: eqx.nn.Linear         # 2*n_hidden → d_model (sin+cos encoding)
    assignment_logits: jnp.ndarray   # (K, n_tokens)

    def __call__(self, theta: jnp.ndarray) -> jnp.ndarray:
        """theta: (K, n_hidden) phases → (n_tokens, d_model) conditioning."""
        encoded = jnp.concatenate([jnp.sin(theta), jnp.cos(theta)], axis=-1)  # (K, 2*n_hidden)
        agent_bias = jax.vmap(self.w_belief)(encoded)
        assignment = jax.nn.softmax(self.assignment_logits, axis=-1)
        return assignment.T @ agent_bias
```
