```python
"""Lorentz hyperboloid operations — pure functions, no parameters."""
from __future__ import annotations
import jax.numpy as jnp


def lorentz_inner(x: jnp.ndarray, y: jnp.ndarray, kappa: float) -> jnp.ndarray:
    """Minkowski inner product: -x_0*y_0 + x_1*y_1 + ... + x_d*y_d."""
    return -x[0] * y[0] + jnp.dot(x[1:], y[1:])


def lorentz_distance(x: jnp.ndarray, y: jnp.ndarray, kappa: float) -> jnp.ndarray:
    """Geodesic distance on the Lorentz hyperboloid."""
    inner = -kappa * lorentz_inner(x, y, kappa)
    return jnp.arccosh(jnp.clip(inner, 1.0 + 1e-7, None)) / jnp.sqrt(kappa)


def exp_map(x: jnp.ndarray, v: jnp.ndarray, kappa: float) -> jnp.ndarray:
    """Exponential map: tangent vector v at point x → point on hyperboloid."""
    v_norm = jnp.sqrt(jnp.clip(kappa * lorentz_inner(v, v, kappa), 1e-8, None))
    return jnp.cosh(v_norm) * x + jnp.sinh(v_norm) * v / (v_norm + 1e-8)


def log_map(x: jnp.ndarray, y: jnp.ndarray, kappa: float) -> jnp.ndarray:
    """Logarithmic map: point y → tangent vector at x."""
    alpha = -kappa * lorentz_inner(x, y, kappa)
    alpha = jnp.clip(alpha, 1.0 + 1e-7, None)
    coeff = jnp.arccosh(alpha) / jnp.sqrt(jnp.clip(alpha ** 2 - 1.0, 1e-8, None))
    return coeff * (y - alpha * x)
```

```python
"""LorentzEmbedding — projects tokens onto Lorentz hyperboloid."""
from __future__ import annotations
import jax
import jax.numpy as jnp
import equinox as eqx
from halo2.config import Halo2Config


class LorentzEmbedding(eqx.Module):
    x_proj: eqx.nn.Linear   # d_model → d_boundary - 1 (spatial components)
    log_curvature: jnp.ndarray  # learnable scalar

    def __init__(self, cfg: Halo2Config, key: jnp.ndarray) -> None:
        self.x_proj = eqx.nn.Linear(cfg.d_model, cfg.d_boundary - 1, key=key)
        self.log_curvature = jnp.log(jnp.array(cfg.init_curvature))

    @property
    def curvature(self) -> jnp.ndarray:
        return jnp.exp(self.log_curvature)

    def __call__(self, h: jnp.ndarray):
        """Args: h (n_tokens, d_model). Returns: x (n_tokens, d_boundary), z (n_tokens,)."""
        kappa = self.curvature
        x_spatial = jax.vmap(self.x_proj)(h)                    # (n_tokens, d_boundary-1)
        x_0 = jnp.sqrt(1.0 / kappa + jnp.sum(x_spatial ** 2, axis=-1, keepdims=True))
        x = jnp.concatenate([x_0, x_spatial], axis=-1)          # (n_tokens, d_boundary)
        z = jax.nn.softplus(x_0[:, 0] - 1.0 / jnp.sqrt(kappa)) # (n_tokens,)
        return x, z
```

```python
"""SelectiveSSM — S7-style input-dependent SSM with parallel scan."""
from __future__ import annotations
import jax
import jax.numpy as jnp
import equinox as eqx
from halo2.config import Halo2Config


class SelectiveSSM(eqx.Module):
    """S7 selective SSM with input-dependent gating and associative scan."""

    A: jnp.ndarray            # (d_state,) diagonal state decay (log scale)
    D: jnp.ndarray            # (d_model,) skip connection
    W_gate: eqx.nn.Linear    # d_model → d_state
    W_dt: eqx.nn.Linear      # d_model → d_state
    W_B: eqx.nn.Linear       # d_model → d_state
    W_C: eqx.nn.Linear       # d_state → d_model

    def __init__(self, cfg: Halo2Config, key: jnp.ndarray) -> None:
        k1, k2, k3, k4 = jax.random.split(key, 4)
        self.A = jnp.full((cfg.d_state,), -1.0)
        self.D = jnp.ones(cfg.d_model)
        self.W_gate = eqx.nn.Linear(cfg.d_model, cfg.d_state, use_bias=False, key=k1)
        self.W_dt = eqx.nn.Linear(cfg.d_model, cfg.d_state, use_bias=False, key=k2)
        self.W_B = eqx.nn.Linear(cfg.d_model, cfg.d_state, use_bias=False, key=k3)
        self.W_C = eqx.nn.Linear(cfg.d_state, cfg.d_model, use_bias=False, key=k4)

    def __call__(self, xs: jnp.ndarray) -> jnp.ndarray:
        """Args: xs (seq_len, d_model). Returns: ys (seq_len, d_model)."""
        gates = jax.nn.sigmoid(jax.vmap(self.W_gate)(xs))       # (T, d_state)
        dt = jax.nn.softplus(jax.vmap(self.W_dt)(xs))           # (T, d_state)
        B_t = jax.vmap(self.W_B)(xs)                             # (T, d_state)

        A_bar = jnp.exp(self.A[None, :] * dt)                   # (T, d_state)
        gated_A = gates * A_bar                                   # (T, d_state)
        gated_B = (1.0 - gates) * B_t                            # (T, d_state)

        def step(h, inputs):
            a, b = inputs
            h_new = a * h + b
            return h_new, h_new

        _, hs = jax.lax.scan(step, jnp.zeros_like(gated_A[0]), (gated_A, gated_B))

        ys = jax.vmap(self.W_C)(hs) + self.D[None, :] * xs      # (T, d_model)
        return ys
```

```python
"""SharedHoloAttention — Zamba2-style shared AdS/CFT attention with LoRA."""
from __future__ import annotations
import jax
import jax.numpy as jnp
import equinox as eqx
from halo2.config import Halo2Config
from halo2.lorentz_ops import lorentz_distance

EPS = 1e-6


class LoRAAdapter(eqx.Module):
    """Per-position LoRA adapter for v_proj and out_proj."""
    A_v: jnp.ndarray     # (d_model, rank)
    B_v: jnp.ndarray     # (rank, n_heads * d_head)
    A_out: jnp.ndarray   # (n_heads * d_head, rank)
    B_out: jnp.ndarray   # (rank, d_model)

    def __init__(self, cfg: Halo2Config, key: jnp.ndarray) -> None:
        k1, k2, k3, k4 = jax.random.split(key, 4)
        r = cfg.lora_rank
        hd = cfg.n_heads * cfg.d_head
        self.A_v = jax.random.normal(k1, (cfg.d_model, r)) * 0.01
        self.B_v = jnp.zeros((r, hd))
        self.A_out = jax.random.normal(k3, (hd, r)) * 0.01
        self.B_out = jnp.zeros((r, cfg.d_model))


class SharedHoloAttention(eqx.Module):
    """Shared holographic attention block with Lorentz kernel."""
    log_delta: jnp.ndarray   # (n_heads,) per-head conformal dimension
    v_proj: eqx.nn.Linear    # d_model → n_heads * d_head
    out_proj: eqx.nn.Linear  # n_heads * d_head → d_model
    n_heads: int = eqx.field(static=True)
    d_head: int = eqx.field(static=True)

    def __init__(self, cfg: Halo2Config, key: jnp.ndarray) -> None:
        k1, k2 = jax.random.split(key)
        self.log_delta = jnp.zeros(cfg.n_heads)
        self.v_proj = eqx.nn.Linear(cfg.d_model, cfg.n_heads * cfg.d_head, key=k1)
        self.out_proj = eqx.nn.Linear(cfg.n_heads * cfg.d_head, cfg.d_model, key=k2)
        self.n_heads = cfg.n_heads
        self.d_head = cfg.d_head

    def __call__(
        self,
        h: jnp.ndarray,           # (N_tok, d_model)
        x: jnp.ndarray,           # (N_tok, d_boundary) Lorentz coords
        z: jnp.ndarray,           # (N_tok,) radial depth
        delta_x: jnp.ndarray | None = None,
        lora: LoRAAdapter | None = None,
    ) -> jnp.ndarray:
        N_tok = h.shape[0]

        # Value projection + LoRA
        V = jax.vmap(self.v_proj)(h)                             # (N_tok, n_heads*d_head)
        if lora is not None:
            V = V + h @ lora.A_v @ lora.B_v
        V_heads = V.reshape(N_tok, self.n_heads, self.d_head).transpose(1, 0, 2)

        # Lorentz distance matrix
        x_b = x if delta_x is None else x + delta_x  # approximate shift in ambient space
        d_geo = jax.vmap(
            lambda xi: jax.vmap(lambda xj: lorentz_distance(xi, xj, 1.0))(x_b)
        )(x_b)  # (N_tok, N_tok)

        base = 1.0 / (jnp.cosh(d_geo) + 1.0 + EPS)             # (N_tok, N_tok)

        def head_attn(log_d, v_h):
            K = base ** jnp.exp(log_d)
            A = jax.nn.softmax(K, axis=-1)
            return A @ v_h

        head_outs = jax.vmap(head_attn)(self.log_delta, V_heads)
        concat = head_outs.transpose(1, 0, 2).reshape(N_tok, self.n_heads * self.d_head)

        out = jax.vmap(self.out_proj)(concat)
        if lora is not None:
            out = out + concat @ lora.A_out @ lora.B_out
        return out
```

```python
"""Halo2Backbone — 24-layer SSSSSH×4 with gradient checkpointing."""
from __future__ import annotations
from typing import Optional
import jax
import jax.numpy as jnp
import equinox as eqx
from halo2.config import Halo2Config
from halo2.ssm_s7 import SelectiveSSM
from halo2.swiglu import SwiGLU
from halo2.holo_attention_shared import SharedHoloAttention, LoRAAdapter
from halo2.step_embed import StepSizeEmbed


class Halo2Backbone(eqx.Module):
    layers: list          # [SelectiveSSM | None] per layer (None = attention position)
    norms1: list          # LayerNorm before core
    norms2: list          # LayerNorm before FFN
    ffns: list            # SwiGLU per layer
    shared_attns: list    # [SharedHoloAttention] × n_shared_attn
    lora_adapters: list   # [LoRAAdapter] per attention position
    step_embed: StepSizeEmbed
    layer_types: tuple = eqx.field(static=True)
    attn_block_ids: tuple = eqx.field(static=True)  # which shared block at each attn pos

    def __init__(self, cfg: Halo2Config, key: jnp.ndarray) -> None:
        pattern = cfg.layer_pattern
        n_repeats = cfg.n_layers // len(pattern)
        self.layer_types = tuple(pattern * n_repeats)

        keys = jax.random.split(key, cfg.n_layers * 3 + cfg.n_shared_attn + 10)
        ki = 0

        self.layers = []
        self.norms1 = []
        self.norms2 = []
        self.ffns = []
        self.lora_adapters = []

        attn_pos_idx = 0
        attn_block_ids = []

        for i, lt in enumerate(self.layer_types):
            if lt == "S":
                self.layers.append(SelectiveSSM(cfg, keys[ki]))
            else:
                self.layers.append(None)  # placeholder — uses shared_attns
                attn_block_ids.append(attn_pos_idx % cfg.n_shared_attn)
                self.lora_adapters.append(LoRAAdapter(cfg, keys[ki]))
                attn_pos_idx += 1
            ki += 1
            self.norms1.append(eqx.nn.LayerNorm(cfg.d_model))
            self.norms2.append(eqx.nn.LayerNorm(cfg.d_model))
            self.ffns.append(SwiGLU(cfg, keys[ki]))
            ki += 1

        self.attn_block_ids = tuple(attn_block_ids)

        self.shared_attns = []
        for j in range(cfg.n_shared_attn):
            self.shared_attns.append(SharedHoloAttention(cfg, keys[ki]))
            ki += 1

        self.step_embed = StepSizeEmbed(cfg, keys[ki])

    def __call__(
        self,
        h: jnp.ndarray,                           # (N_tok, d_model)
        x: jnp.ndarray,                           # (N_tok, d_boundary)
        z: jnp.ndarray,                           # (N_tok,)
        delta_x: Optional[jnp.ndarray] = None,
        step_size: float = 1.0,
    ) -> jnp.ndarray:
        s_emb = self.step_embed(step_size)  # (d_model,)
        h = h + s_emb[None, :]              # broadcast over tokens

        attn_idx = 0
        for i, lt in enumerate(self.layer_types):
            norm1 = self.norms1[i]
            norm2 = self.norms2[i]
            ffn = self.ffns[i]

            @jax.checkpoint
            def layer_fn(h_in, _lt=lt, _i=i, _attn_idx=attn_idx):
                h_n = jax.vmap(norm1)(h_in)
                if _lt == "S":
                    h_core = self.layers[_i](h_n)
                else:
                    block_id = self.attn_block_ids[_attn_idx]
                    lora = self.lora_adapters[_attn_idx]
                    h_core = self.shared_attns[block_id](
                        h_n, x, z, delta_x=delta_x, lora=lora
                    )
                h_out = h_in + h_core
                h_out = h_out + jax.vmap(ffn)(jax.vmap(norm2)(h_out))
                return h_out

            h = layer_fn(h)
            if lt == "H":
                attn_idx += 1

        return h
```

```python
"""DiscreteGM — categorical generative model with A, B, C, D matrices."""
from __future__ import annotations
import jax
import jax.numpy as jnp
import equinox as eqx
from halo2.config import Halo2Config


class DiscreteGM(eqx.Module):
    log_A: jnp.ndarray  # (n_obs, n_hidden)
    log_B: jnp.ndarray  # (n_hidden, n_hidden, n_actions)
    log_C: jnp.ndarray  # (n_obs,)
    log_D: jnp.ndarray  # (n_hidden,)

    def __init__(self, cfg: Halo2Config, key: jnp.ndarray) -> None:
        k1, k2, k3, k4 = jax.random.split(key, 4)
        self.log_A = jax.random.normal(k1, (cfg.n_obs, cfg.n_hidden))
        self.log_B = jax.random.normal(k2, (cfg.n_hidden, cfg.n_hidden, cfg.n_actions))
        self.log_C = jax.random.normal(k3, (cfg.n_obs,))
        self.log_D = jax.random.normal(k4, (cfg.n_hidden,))

    @property
    def A(self) -> jnp.ndarray:
        return jax.nn.softmax(self.log_A, axis=0)
```

```python
    For categorical models: posterior ∝ prior × likelihood.
    In log space: log_posterior = log_prior + log_likelihood.
```

```python
"""DPEFE — Dynamic Programming Expected Free Energy via backward Bellman."""
from __future__ import annotations
import jax
import jax.numpy as jnp
from halo2.config import Halo2Config
from halo2.fep.generative_model import DiscreteGM


def dpefe_action_selection(
    mu: jnp.ndarray,
    gm: DiscreteGM,
    cfg: Halo2Config,
) -> jnp.ndarray:
    """Backward Bellman EFE. Returns (n_actions,) action probabilities."""
    q_eta = jax.nn.softmax(mu)
    p_obs = jax.nn.softmax(gm.log_C)
    H_s = -jnp.sum(gm.A * jnp.log(gm.A + 1e-8), axis=0)  # (n_hidden,)

    V = jnp.zeros(cfg.n_hidden)

    def backward_step(V, _):
        def q_value(a):
            q_next = gm.B[:, :, a] @ q_eta
            q_obs = gm.A @ q_next
            pragmatic = -jnp.sum(q_obs * (jnp.log(q_obs + 1e-8) - jnp.log(p_obs + 1e-8)))
            epistemic = jnp.sum(q_next * H_s)
            future = jnp.sum(q_next * V)
            return pragmatic + epistemic + future

        Q = jax.vmap(q_value)(jnp.arange(cfg.n_actions))
        V_new = jax.nn.softmax(-cfg.beta * Q) @ Q
        return V_new, Q

    _, Q_all = jax.lax.scan(backward_step, V, None, length=cfg.tau)
    return jax.nn.softmax(-cfg.beta * Q_all[0])
```

```python
"""AdS Klein-Gordon flow prior — Lorentz-native, boundary space."""
from __future__ import annotations
import jax
import jax.numpy as jnp
from halo2.lorentz_ops import lorentz_distance, log_map


def ads_kg_prior_lorentz(
    x_noise: jnp.ndarray,   # (N_tok, d_boundary) Lorentz coords
    x_data: jnp.ndarray,    # (N_tok, d_boundary) Lorentz coords
    t: float,
    delta_flow: float,
    kappa: float,
) -> jnp.ndarray:
    """Lorentz-native KG prior. Returns (N_tok, d_boundary) tangent vectors."""
    z_t = 1.0 - t + 1e-6
    d_geo = jax.vmap(
        lambda xn, xd: lorentz_distance(xn, xd, kappa)
    )(x_noise, x_data)                                       # (N_tok,)
    K = (z_t / (z_t ** 2 + jnp.sinh(d_geo) ** 2 + 1e-6)) ** delta_flow
    weights = jax.nn.softmax(K)                               # (N_tok,)
    tangents = jax.vmap(
        lambda xn, xd: log_map(xn, xd, kappa)
    )(x_noise, x_data)                                       # (N_tok, d_boundary)
    return weights[:, None] * tangents
```

```python
from typing import NamedTuple
import jax
import jax.numpy as jnp
import equinox as eqx
from halo2.config import Halo2Config


class PageMemState(NamedTuple):
    cache: jnp.ndarray
    n_cached: jnp.ndarray
    island: jnp.ndarray
    island_ptr: jnp.ndarray


class PageCurveMemory(eqx.Module):
    max_cache: int = eqx.field(static=True)
    island_size: int = eqx.field(static=True)
    d_model: int = eqx.field(static=True)
    d_head: int = eqx.field(static=True)

    def __init__(self, cfg: Halo2Config) -> None:
        self.max_cache = cfg.max_cache
        self.island_size = cfg.island_size
        self.d_model = cfg.d_model
        self.d_head = cfg.d_head

    def init_state(self) -> PageMemState:
        return PageMemState(
            cache=jnp.zeros((self.max_cache, self.d_model)),
            n_cached=jnp.int32(0),
            island=jnp.zeros((self.island_size, self.d_model)),
            island_ptr=jnp.int32(0),
        )

    def __call__(self, x_i: jnp.ndarray, state: PageMemState) -> PageMemState:
        write_ptr = state.n_cached % self.max_cache
        new_cache = state.cache.at[write_ptr].set(x_i)
        is_full = state.n_cached >= self.max_cache
        s_gen = jnp.sum(new_cache ** 2, axis=-1) * self.d_head / 4.0
        evict_idx = jnp.argmin(s_gen)
        evicted = new_cache[evict_idx]
        iptr = state.island_ptr % self.island_size
        new_island = jnp.where(is_full, state.island.at[iptr].set(evicted), state.island)
        new_iptr = jnp.where(is_full, state.island_ptr + 1, state.island_ptr)
        return PageMemState(
            cache=new_cache, n_cached=state.n_cached + 1,
            island=new_island, island_ptr=new_iptr,
        )
```

```python
from __future__ import annotations
import jax
import jax.numpy as jnp
from halo2.config import Halo2Config


def unified_elbo_loss_v2(model, carry, tokens, key):
    """L = L_flow + L_obs + L_prior + physics regularizers.

    Returns (total_loss, aux_dict).
    """
    from halo2.model import halo2_step

    new_carry, (h_out, obs, v_pred, v_target) = halo2_step(model, carry, tokens, key)

    # Flow matching in boundary space
    l_flow = jnp.mean((v_pred - v_target) ** 2)

    # Observation likelihood over K clusters
    q_eta = jax.nn.softmax(new_carry.swarm.cluster_mu)
    log_A = jnp.log(model.gm.A + 1e-8)
    l_obs = -jnp.mean(jnp.einsum('ko,oi,ki->k', obs, log_A, q_eta))

    # KL prior
    log_q = jnp.log(q_eta + 1e-8)
    log_D = jnp.log(model.gm.D + 1e-8)
    l_prior = jnp.mean(jnp.sum(q_eta * (log_q - log_D), axis=-1))

    total = l_flow + l_obs + l_prior
    return total, {"l_flow": l_flow, "l_obs": l_obs, "l_prior": l_prior}
```

```python
        self.cfg = cfg
        self.lorentz_embed = LorentzEmbedding(cfg, keys[0])
        self.backbone = Halo2Backbone(cfg, keys[1])
        self.obs_bridge = ObsBridge(cfg, keys[2])
        self.action_bridge = ActionBridge(cfg, keys[3])
        self.belief_bridge = BeliefBridge(cfg, keys[4])
        self.page_memory = PageCurveMemory(cfg)
        self.gm = DiscreteGM(cfg, keys[5])
        self.v_proj = eqx.nn.Linear(cfg.d_boundary, cfg.d_model, use_bias=False, key=keys[6])

    def init_carry(self, key: jnp.ndarray) -> Halo2Carry:
        return Halo2Carry(
            swarm=init_swarm(self.cfg, key),
            page_mem=self.page_memory.init_state(),
            key=key,
        )


def halo2_step(
    model: Halo2Model,
    carry: Halo2Carry,
    tokens: jnp.ndarray,
    key: jnp.ndarray,
) -> tuple:
    """One closed-loop step. Returns (new_carry, (h_out, obs, v_pred, v_target))."""
    cfg = model.cfg
    k1, k2, k3 = jax.random.split(key, 3)

    # FEP → HALO conditioning
    delta_x = model.action_bridge(carry.swarm.cluster_action)
    delta_v = model.belief_bridge(carry.swarm.cluster_mu)

    # Lorentz embedding
    x_data, z_data = model.lorentz_embed(tokens)

    # Flow matching in boundary space
    t = jax.random.uniform(k1)
    x_noise_raw = jax.random.normal(k2, x_data[:, 1:].shape)
    x_noise_0 = jnp.sqrt(1.0 / cfg.init_curvature + jnp.sum(x_noise_raw ** 2, axis=-1, keepdims=True))
    x_noise = jnp.concatenate([x_noise_0, x_noise_raw], axis=-1)

    # Geodesic interpolation (simplified: linear in ambient + reproject)
    x_t = (1.0 - t) * x_noise + t * x_data
    x_t_0 = jnp.sqrt(1.0 / cfg.init_curvature + jnp.sum(x_t[:, 1:] ** 2, axis=-1, keepdims=True))
    x_t = x_t.at[:, :1].set(x_t_0)

    # Backbone in d_model space
    h_t = jax.vmap(model.v_proj)(x_t) + delta_v
    step_size = jax.random.uniform(k3) if cfg.shortcut else 1.0
    h_out = model.backbone(h_t, x_data, z_data, delta_x=delta_x, step_size=float(step_size))

    # Flow targets in boundary space
    v_target = x_data - x_noise
    v_kg = ads_kg_prior_lorentz(x_noise, x_data, t, cfg.delta_flow, cfg.init_curvature)
    # Project backbone output back to boundary
    x_proj_weight = model.lorentz_embed.x_proj.weight  # (d_boundary-1, d_model)
    h_boundary = h_out @ x_proj_weight.T  # (N_tok, d_boundary-1)
    h_boundary_0 = jnp.zeros((h_boundary.shape[0], 1))
    v_pred = jnp.concatenate([h_boundary_0, h_boundary], axis=-1) + v_kg

    # Page memory
    def update_mem(mem, x_i):
        return model.page_memory(x_i, mem), None
    new_page_mem, _ = jax.lax.scan(update_mem, carry.page_mem, h_out)

    # HALO → FEP observations
    obs = model.obs_bridge(h_out)

    # FEP swarm step
    new_swarm = swarm_step(carry.swarm, obs, model.gm, cfg)

    new_carry = Halo2Carry(
        swarm=new_swarm,
        page_mem=new_page_mem,
        key=jax.random.split(key)[0],
    )
    return new_carry, (h_out, obs, v_pred, v_target)
```

```python
"""Halo2Trainer — Adafactor + FP16 + gradient checkpointing + buffer donation."""
from __future__ import annotations
import logging
from typing import Any
import jax
import jax.numpy as jnp
import jax.tree_util as jtu
import equinox as eqx
import optax
import numpy as np
from halo2.config import Halo2Config
from halo2.model import Halo2Model, halo2_step
from halo2.loss import unified_elbo_loss_v2

log = logging.getLogger(__name__)


class Halo2Trainer:
    def __init__(self, cfg: Halo2Config, n_steps: int = 100, lr: float = 3e-4) -> None:
        self.cfg = cfg
        self.n_steps = n_steps
        schedule = optax.warmup_cosine_decay_schedule(
            init_value=0.0, peak_value=lr,
            warmup_steps=min(50, n_steps // 10),
            decay_steps=n_steps, end_value=lr * 0.1,
        )
        self.opt = optax.adafactor(learning_rate=schedule)

    def run(self, model: Halo2Model, episodes: list) -> tuple[Halo2Model, dict[str, Any]]:
        if not episodes:
            return model, {"loss_before": 0.0, "loss_after": 0.0, "n_episodes": 0}

        key = jax.random.PRNGKey(self.cfg.seed)
        carry = model.init_carry(key)
        opt_state = self.opt.init(eqx.filter(model, eqx.is_array))

        loss_before = self._mean_loss(model, episodes, key)

        for step in range(self.n_steps):
            ep = episodes[step % len(episodes)]
            tokens = jnp.array(ep.tokens)
            key, sk = jax.random.split(key)

            loss_fn = lambda m: unified_elbo_loss_v2(m, carry, tokens, sk)[0]
            loss, grads = eqx.filter_value_and_grad(loss_fn)(model)
            updates, opt_state = self.opt.update(
                eqx.filter(grads, eqx.is_array), opt_state,
                eqx.filter(model, eqx.is_array),
            )
            model = eqx.apply_updates(model, updates)

            if step % 100 == 0:
                log.info(f"Step {step}/{self.n_steps} | loss={float(loss):.4f}")

        loss_after = self._mean_loss(model, episodes, key)
        return model, {
            "loss_before": float(loss_before),
            "loss_after": float(loss_after),
            "n_episodes": len(episodes),
        }

    def _mean_loss(self, model, episodes, key):
        carry = model.init_carry(key)
        losses = []
        for ep in episodes[:10]:
            tokens = jnp.array(ep.tokens)
            key, sk = jax.random.split(key)
            loss, _ = unified_elbo_loss_v2(model, carry, tokens, sk)
            losses.append(float(loss))
        return float(np.mean(losses))
```

```python
"""Phase 0 bootstrap pre-training for HoloBiont 2.0."""
from __future__ import annotations
import logging
import os
import jax
import jax.numpy as jnp
import equinox as eqx
import optax
from halo2.config import Halo2Config
from halo2.model import Halo2Model, halo2_step
from halo2.loss import unified_elbo_loss_v2

log = logging.getLogger(__name__)


def save_checkpoint(model: Halo2Model, path: str) -> None:
    os.makedirs(os.path.dirname(path) if os.path.dirname(path) else ".", exist_ok=True)
    eqx.tree_serialise_leaves(path + ".eqx", model)
    log.info(f"Checkpoint saved to {path}.eqx")


def load_checkpoint(cfg: Halo2Config, path: str) -> Halo2Model:
    template = Halo2Model(cfg, jax.random.PRNGKey(0))
    return eqx.tree_deserialise_leaves(path + ".eqx", template)


def run_bootstrap(
    cfg: Halo2Config,
    n_steps: int = 5000,
    checkpoint_dir: str = "data/checkpoints/halo2",
    seed: int = 42,
) -> Halo2Model:
    key = jax.random.PRNGKey(seed)
    k1, k2 = jax.random.split(key)
    model = Halo2Model(cfg, k1)
    carry = model.init_carry(key)

    schedule = optax.warmup_cosine_decay_schedule(
        init_value=0.0, peak_value=cfg.lr,
        warmup_steps=500, decay_steps=n_steps, end_value=cfg.lr * 0.1,
    )
    opt = optax.adafactor(learning_rate=schedule)
    opt_state = opt.init(eqx.filter(model, eqx.is_array))

    for step in range(n_steps):
        key, sk1, sk2 = jax.random.split(key, 3)
        tokens = jax.random.normal(sk1, (cfg.n_tokens, cfg.d_model))

        loss_fn = lambda m: unified_elbo_loss_v2(m, carry, tokens, sk2)[0]
        loss, grads = eqx.filter_value_and_grad(loss_fn)(model)
        updates, opt_state = opt.update(
            eqx.filter(grads, eqx.is_array), opt_state,
            eqx.filter(model, eqx.is_array),
        )
        model = eqx.apply_updates(model, updates)

        if step % 500 == 0:
            log.info(f"Step {step}/{n_steps} | loss={float(loss):.4f}")

    save_checkpoint(model, checkpoint_dir)
    return model


if __name__ == "__main__":
    logging.basicConfig(level=logging.INFO)
    cfg = Halo2Config()
    run_bootstrap(cfg)
```
